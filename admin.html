Cloudinary settings
Free account at cloudinary.com. Go to Settings → Upload → Upload presets → add one set to Unsigned. Saved in your browser only.
Step 1 — GitHub personal access token
1. Go to github.com/settings/tokens/new.
2. Name it site-admin, set an expiry (90 days is fine), tick the repo scope.
3. Generate and copy it — you only see it once.
4. Paste it into the login screen. It's saved in localStorage on your device only, never sent anywhere except GitHub's own API.
Step 2 — Change the admin password
Open admin.html, find the line near the top of the script that reads const ADMIN_PASSWORD = "change-me" and update it. Do not commit this file to a public repo if you care about the password — or just accept that it's a personal site and it's fine.
Step 3 — Create content.json in your repo
On GitHub, in your site repo, create a new file called content.json and paste in the JSON below. After that, the admin panel takes over — you never edit it by hand again.


    

      
Step 4 — Hook up your HTML pages

      

        
Add id attributes to the elements you want to update, then paste the snippet below before </body> on each page. The page falls back gracefully to its hardcoded content if content.json is missing.

      

    


    
<script>
fetch('content.json')
  .then(r => r.json())
  .then(data => {
    // Helper: set text if element exists and value is non-empty
    const t = (id, val) => { const el = document.getElementById(id); if (el && val) el.textContent = val; };
    // Helper: set background image
    const bg = (id, url) => { const el = document.getElementById(id); if (el && url) el.style.backgroundImage = `url(${url})`; };
    // Helper: set img src
    const img = (id, url) => { const el = document.getElementById(id); if (el && url) el.src = url; };

    // --- index.html ---
    t('bio-p1', data.home.bio_p1);
    t('bio-p2', data.home.bio_p2);
    t('bio-p3', data.home.bio_p3);
    t('fact-from', data.home.fact_from);
    t('fact-based', data.home.fact_based);
    t('fact-currently', data.home.fact_currently);
    t('fact-education', data.home.fact_education);
    t('fact-interests', data.home.fact_interests);
    img('photo-portrait', data.photos.home_portrait);

    // --- life.html ---
    t('hobby-climbing', data.life.hobby_climbing_desc);
    t('hobby-bouldering', data.life.hobby_bouldering_desc);
    t('hobby-martial', data.life.hobby_martial_desc);
    t('hobby-reading', data.life.hobby_reading_desc);
    t('hobby-travel', data.life.hobby_travel_desc);
    t('hobby-energy', data.life.hobby_energy_desc);
    t('book-1-title', data.life.book_1_title);
    t('book-1-author', data.life.book_1_author);
    t('book-2-title', data.life.book_2_title);
    t('book-2-author', data.life.book_2_author);
    t('book-3-title', data.life.book_3_title);
    t('book-3-author', data.life.book_3_author);

    // --- travel.html ---
    t('travel-hero-sub', data.travel.hero_sub);
    t('travel-namerica', data.travel.namerica_text);
    t('travel-asia', data.travel.asia_text);
    t('travel-australia', data.travel.australia_text);
    t('travel-japan', data.travel.japan_text);
    t('travel-africa', data.travel.africa_text);
    t('travel-europe', data.travel.europe_text);
    t('travel-samerica', data.travel.samerica_text);

    // --- africa.html ---
    bg('photo-casa-1', data.photos.africa_casablanca_1);
    bg('photo-casa-2', data.photos.africa_casablanca_2);
    bg('photo-marra-1', data.photos.africa_marrakech_1);
    bg('photo-marra-2', data.photos.africa_marrakech_2);
    bg('photo-marra-3', data.photos.africa_marrakech_3);
  })
  .catch(() => { /* content.json missing — hardcoded content shows instead */ });
</script>
